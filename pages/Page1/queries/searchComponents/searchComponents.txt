SELECT 
    c.id, c.part_number, c.value_display, c.package,
    c.tolerance_percent, c.voltage_rating_v, c.temp_min_c, c.temp_max_c,
    c.status, cat.name AS category_name, m.name AS manufacturer_name,
    c.library_path, c.library_ref, c.footprint_path, c.footprint_ref
FROM components c
LEFT JOIN categories cat ON c.category_id = cat.id
LEFT JOIN manufacturers m ON c.manufacturer_id = m.id
WHERE 
    ({{searchInput.text}} = '' OR c.part_number ILIKE '%{{searchInput.text}}%')
    AND ({{filterCategorySelect.selectedOptionValue}} = '' OR c.category_id = {{filterCategorySelect.selectedOptionValue}})
    AND ({{filterPackageSelect.selectedOptionValue}} = '' OR c.package = {{filterPackageSelect.selectedOptionValue}})
ORDER BY c.part_number;
