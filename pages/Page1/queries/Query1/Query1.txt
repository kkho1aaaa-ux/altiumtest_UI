-- Добавляем новые поля в таблицу categories
ALTER TABLE categories 
    ADD COLUMN IF NOT EXISTS schlib_path VARCHAR(255),
    ADD COLUMN IF NOT EXISTS pcblib_path VARCHAR(255),
    ADD COLUMN IF NOT EXISTS altium_designator VARCHAR(50);

-- Проверяем, что поля добавлены
SELECT column_name, data_type 
FROM information_schema.columns 
WHERE table_name = 'categories'
ORDER BY ordinal_position;
