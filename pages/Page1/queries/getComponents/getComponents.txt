SELECT 
    c.id,
    c.part_number,
    c.category_id,
    c.manufacturer_id,
    c.library_path,
    c.library_ref,
    c.footprint_path,
    c.footprint_ref,
    c.datasheet_url,
    c.spice_model_path,
    c.value_numeric,
    c.value_unit,
    c.value_display,
    c.tolerance_percent,
    c.voltage_rating_v,
    c.power_rating_w,
    c.temp_min_c,
    c.temp_max_c,
    c.resistance_ohm,
    c.capacitance_pf,
    c.inductance_uh,
    c.package,
    c.altium_comment,
    c.altium_designator,
    c.kicad_keywords,
    c.status,
    cat.name AS category_name,
    m.name AS manufacturer_name
FROM components c
LEFT JOIN categories cat ON c.category_id = cat.id
LEFT JOIN manufacturers m ON c.manufacturer_id = m.id
ORDER BY c.part_number;
