SELECT 
    c.id,
    c.part_number,
    c.category_id,
    cat.name as category_name,
    c.manufacturer_id,
    m.name as manufacturer_name,
    c.package,
    c.package_id,
    ps.name as package_standard_name,
    c.value_number,
    c.value_multiplier,
    c.value_unit,
    c.value_display,
    c.tolerance_percent,
    c.voltage_rating_v,
    c.power_rating_w,
    c.temp_min_c,
    c.temp_max_c,
    c.library_path,
    c.library_ref,
    c.footprint_path,
    c.footprint_ref,
    c.datasheet_url,
    c.spice_model_path,
    c.altium_comment,
    c.altium_designator,
    c.kicad_keywords,
    c.kicad_fp_filter
FROM components c
LEFT JOIN categories cat ON c.category_id = cat.id
LEFT JOIN manufacturers m ON c.manufacturer_id = m.id
LEFT JOIN packages pkg ON c.package_id = pkg.id
LEFT JOIN package_standards ps ON pkg.standard_id = ps.id
ORDER BY c.id DESC
