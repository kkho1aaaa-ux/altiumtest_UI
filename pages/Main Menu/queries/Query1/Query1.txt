-- VIEW для конденсаторов
CREATE OR REPLACE VIEW v_kicad_capacitors AS
SELECT 
    c.id,
    c.part_number,
    cat.name AS category_name,
    c.value_display,
    c.value_numeric,
    c.value_unit,
    c.tolerance_percent,
    c.voltage_rating_v,
    c.power_rating_w,
    c.temp_min_c,
    c.temp_max_c,
    c.package,
    c.package_standard,
    m.name AS manufacturer_name,
    c.library_path,
    c.library_ref,
    c.footprint_path,
    c.footprint_ref,
    c.datasheet_url,
    c.spice_model_path,
    c.altium_comment,
    c.altium_designator,
    c.kicad_keywords,
    c.kicad_fp_filter,
    -- Уникальное имя символа для KiCad
    CONCAT('C_', c.part_number) AS symbol_name,
    'Device' AS symbol_lib_name
FROM components c
LEFT JOIN categories cat ON c.category_id = cat.id
LEFT JOIN manufacturers m ON c.manufacturer_id = m.id
WHERE cat.designator_prefix = 'C';
