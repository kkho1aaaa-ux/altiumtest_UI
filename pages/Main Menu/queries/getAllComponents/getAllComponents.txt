SELECT 
    c.id,
    c.part_number,
    c.category_id,
    cat.name AS category_name,
    CASE cat.name
        WHEN 'Resistors' THEN 'Резисторы'
        WHEN 'Capacitors' THEN 'Конденсаторы'
        WHEN 'Inductors' THEN 'Индуктивности'
        WHEN 'Diodes' THEN 'Диоды'
        WHEN 'LEDs' THEN 'Светодиоды'
        WHEN 'Transistors' THEN 'Транзисторы'
        WHEN 'ICs' THEN 'Микросхемы'
        WHEN 'Connectors' THEN 'Разъёмы'
        ELSE cat.name
    END AS category_name_ru,
    cat.designator_prefix,
    c.manufacturer_id,
    m.name AS manufacturer_name,
    c.value_display,
    c.value_unit,
    c.value_numeric,
    c.tolerance_percent,
    c.temp_min_c,
    c.temp_max_c,
    c.forward_voltage_v,
    c.reverse_voltage_v,
    c.output_voltage_v,
    c.dropout_voltage_v,
    c.pin_count,
    c.pitch_mm,
    c.q_factor,
    c.dielectric_type,
    c.transistor_type,
    c.channel_type,
    c.is_polarized,
    c.package,
    c.package_standard,
    c.library_path,
    c.library_ref,
    c.footprint_path,
    c.footprint_ref,
    c.datasheet_url,
    c.spice_model_path,
    c.altium_comment,
    c.altium_designator,
    c.kicad_keywords,
    c.kicad_fp_filter,
    c.created_at,
    c.updated_at
FROM components c
LEFT JOIN categories cat ON c.category_id = cat.id
LEFT JOIN manufacturers m ON c.manufacturer_id = m.id
ORDER BY c.part_number ASC;
